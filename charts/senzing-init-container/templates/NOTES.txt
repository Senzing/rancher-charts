    ```console
    kubectl get pods --namespace {{ .Release.Namespace }} -l app.kubernetes.io/instance={{ .Release.Name }}
    ```

2. Access the pod you want to debug by executing

    ```console
    kubectl exec --namespace {{ .Release.Namespace }} -ti <NAME OF THE POD> -- bash
    ```
